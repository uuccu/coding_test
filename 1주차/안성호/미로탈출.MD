#include <bits/stdc++.h>
using namespace std;
int dx[4]={1,0,-1,0};
int dy[4]={0,1,0,-1};
int N,M;
int sx,sy,ex,ey,lx,ly;
bool vis[101][101];
bool vis1[101][101];
char m[101][101];

int bfs1(int x,int y);
bool isBound(int x,int y)
{
    if(x>=0&&x<N&&y>=0&&y<M)
        return true;
    else
        return false;
}
int bfs(int x,int y)
{
    queue<pair<pair<int,int>,int>> q;
    q.push({{x,y},0}); //시작 ->레버까지
    int toL=-1;
    while(!q.empty())
    {
        int nx=q.front().first.first;
        int ny=q.front().first.second;
        int cnt=q.front().second;
        q.pop();
        if(m[nx][ny]=='L')
        {
            lx=nx;
            ly=ny;
            toL=cnt;
            break;
        }
        for(int i=0;i<4;i++)
        {
            int tx=nx+dx[i];
            int ty=ny+dy[i];
            if(isBound(tx,ty)&&!vis[tx][ty]&&m[tx][ty]!='X')
            {
                vis[tx][ty]=true;
                q.push({{tx,ty},cnt+1});
            }
        }
    }
    if(toL==-1)
        return -1;
    int toE=bfs1(lx,ly);
    if(toL==-1||toE==-1)
        return -1;
    else
        return toL+toE;
}
int bfs1(int x,int y)
{
    queue<pair<pair<int,int>,int>> q;
    q.push({{x,y},0}); //시작 ->레버까지
    int toE=-1;
    while(!q.empty())
    {
        int nx=q.front().first.first;
        int ny=q.front().first.second;
        int cnt=q.front().second;
        q.pop();
        if(m[nx][ny]=='E')
        {
            return cnt;
        }
        for(int i=0;i<4;i++)
        {
            int tx=nx+dx[i];
            int ty=ny+dy[i];
            if(isBound(tx,ty)&&!vis1[tx][ty]&&m[tx][ty]!='X')
            {
                vis1[tx][ty]=true;
                q.push({{tx,ty},cnt+1});
            }
        }
    }
    return toE;
}
int solution(vector<string> maps) {
    N=maps.size();
    M=maps[0].length();
    int answer = 0;
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<M;j++)
        {
            m[i][j]=maps[i][j];
            if(maps[i][j]=='S') //시작점
            {
                sx=i;
                sy=j;
                continue;
            }
            if(maps[i][j]=='E')
            {
                ex=i;
                ey=j;
                continue;
            }

        }
    }
    return bfs(sx,sy);
}
