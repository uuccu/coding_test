이차원 배열과 연산

```c++
#include <bits/stdc++.h>


using namespace std;

#define MAX 101;
int arr[MAX][MAX];
int r,c,k;
int answer;
int cnt[MAX];


void solve(){

    

    int t = 0;
    int garo = 3;
    int sero = 3;

    while(t <= 100){// 찾아질때까지 무한반복 100초 지나면 -1임.
        if(arr[r][c] == k){// base case
            cout<<time;
            return;
        }

        vector<int> temp;
        if(garo >= sero){
            for(int i = 0; i< garo ; i++){
                vector<pair<int,int>> v;
                memset(cnt,0,sizeof(cnt));
                for(int j = 0; j<sero;j++){
                    cnt[arr[i][j]]++;
                }
                for(int j = 0; j<MAX;j++){
                    if(cnt[j] == 0)continue;
                    v.push_back({cnt[j],j});
                }
                sort(v.begin(),v.end());
                for(int j = 0; j<v.size();j++)arr[i][j] = 0;
                int idx = 0;
                for(int j = 0; j<v.size();j++){
                    arr[i][idx++] = v[j].second;
                    arr[i][idx++] = v[j].first;
                }
                // idx--;
                temp.push_back(idx);
            }

        }else {

        }
        t++;
    }

    return -1;
}

int main() {
    cin>>r>>c>>k;
    answer = 0;
    for(int i=1;i<=3;i++){// 입력
        for(int j=1;j<=3;j++){
            cin>>arr[i][j];
        }
    }
    solve();
}
```
